## 链表相交代码

结合前面的一篇文章，将有环和无环的结合在一起写出代码
```
class Node:
    def __init__(self, val):
        self.val = val
        self.next = None

class Solution:
    def detect(self, head1, head2):
        if not head1 or not head2:
            return None
        circle1 = self.has_circle(head1)
        circle2 = self.has_circle(head2)
        # 都没环，寻找相交节点
        if not circle1 and not circle2:
            return self.get_non_circle_intersection(head1, head2)
        # 都有环，寻找相交点
        elif circle1 and circle2:
            return self.circle_intersection(head1, circle1, head2, circle2)
        else:
            return None

    def get_non_circle_intersection(self, head1, head2):
        p1, p2 = head1, head2
        len1, len2 = 0, 0
        while (p1.next):
            p1 = p1.next
            len1 += 1
        while p2.next:
            p2 = p2.next
            len2 += 1
        # 判断是否相交
        if p1 != p2:
            return None
        diff = abs(len1 - len2)
        if (len1 >= len2):
            p1 = head1
            p2 = head2
        else:
            p1 = head2
            p2 = head1
        while diff > 0:
            p1 = p1.next
            diff -= 1
        while p1 != p2:
            p1 = p1.next
            p2 = p2.next
        return p1

    def circle_intersection(self, head1, circle1, head2, circle2):
        tmp = circle1.next
        while(tmp != circle1):
            if tmp == circle2:
                break
            tmp = tmp.next
        # 判断是否相交
        if tmp != circle2:
            return None
        len1, len2 = 0, 0
        tmp1, tmp2 = head1, head2
        while tmp1 != circle2:
            tmp1 = tmp1.next
            len1 += 1
        while tmp2 != circle2:
            tmp2 = tmp2.next
            len2 += 1
        p1, p2 = head1, head2
        if len1 >= len2:
            diff = len1 - len2
            while diff > 0:
                p1 = p1.next
                diff -= 1
        else:
            diff = len2 - len1
            while diff > 0:
                p2 = p2.next
                diff -= 1

        while(p1 != p2):
            p1 = p1.next
            p2 = p2.next
        return p1

    def has_circle(self, head):
        if not head:
            return False
        fast, slow = head, head
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
            if slow == fast:
                # find the intersection point
                return slow
        return None
# non circle intersect
n1 = Node(1)
n2 = Node(2)
n3 = Node(3)
n4 = Node(4)
n22 = Node(2)
n1.next = n2
n2.next = n3
n3.next = n4
n22.next = n3
head1 = n1
head2 = n22
s = Solution()
print(s.detect(head1, head2).val)

n32.next = n33
n33.next = n34
n34.next = n35
n35.next = n33
n41.next = n42
n42.next = n43
n43.next = n33
head11 = n31
head12 = n41

print(s1.detect(head11, head12).val)
```
